77 bool checkDirection(int idx) {
78 float *current_window = spectrogram->getCurrentWindow();
79 float *prev_window = spectrogram->getPreviousWindow();
80 if (direction == 1 && current_window[idx] > prev_window[idx])
81 return false;
82 if (direction == 2 && current_window[idx] < prev_window[idx])

87 // finds the n (numFreqs) bins with highest change in amplitude, stored in salientFreqs[]
88 // input is assumed to be a 2D array of FFT data passed in from the Vibrosonics class
89 void doAnalysis()

91 deltaAmps.doAnalysis();
92 amplitudes = deltaAmps.getOutput(); // copy amplitudes
93
94 // iterate through amplitudes to find the maximum(s)
95 int currMaxAmp = 0;
96 int currMaxAmpIdx = -1;
97 for (int i = 0; i < numFreqs; i++) {
98 for (int j = lowerBinBound; j < upperBinBound; j++) {
99 if (amplitudes[j] > currMaxAmp) {
100 if(checkDirection(j)){
101 currMaxAmp = amplitudes[j];
102 currMaxAmpIdx = j;
103 }
104 }
105 }
106 salientFreqs[i] = currMaxAmpIdx; // add the new max amp index to the array
107 amplitudes[currMaxAmpIdx] = 0; // set the amp to 0 so it will not be found again
108 currMaxAmp = 0; // reset iterators
109 currMaxAmpIdx = -1;
110 }
111 output = salientFreqs;
