4// Name : Noisiness
5// Return Type : float (noisiness of the current window)
6// Description : Calculates the noisiness of the current window. Noisiness is
7// the opposite of periodicity, so a low noisiness value
8// indicates a high degree of periodicity, like a sine wave, and
9// a high noisiness value indicates a low degree of periodicity,
10// like white noise.
11//
12// The noisiness of the current window is the
13// entropy of the normalized amplitude spectrum. This is a
14// measure of the amount of the randomness / unpredictability in
15// the current window, treating the normalized amplitude
16// spectrum as a probability distribution.
17//============================================================================

55 float entropy = 0;
56 // for each bin, calculate the contribution to the overall entropy
57 for (int i = lowerBinBound; i < upperBinBound; i++) {
58 // if the amplitude of the current bin is zero, skip it (log2(0) is undefined)
59 if (windowData[i] > 0) {
60 // assign a probability to each bin based on its amplitude
61 float p = windowData[i] / total;
62 // calculate the contribution of this bin to the entropy
63 entropy -= p * log2(p);
64 }
65 }
66
67 // Normalize the entropy to a 0.-1. scale
68 output = entropy / log2(upperBinBound - lowerBinBound);
69
