46 void setBands(int* frequencyBands, int numBands)

48 int _nyquist = sampleRate >> 1; // nyquist frequency is 1/2 the sampleRate
49
50 for (int i = 0; i < numBands; i++) { // check if each band is greater than the previous and less than next
51 if (!((frequencyBands[i] >= 0 && frequencyBands[i] <= _nyquist) && (frequencyBands[i] < frequencyBands[i + 1] && frequencyBands[i + 1] <= _nyquist))) {
52 Serial.println("BreadSlicer setBands() fail! Invalid bands!");
53 return;
54 }
55 }

57 // free old bandIndexes and output pointers if bands have already been set

62
63 // allocate memory for new bandIndexes and outpt pointers
64 this->bandIndexes = (int*)malloc(sizeof(int) * numBands + 1);
65 this->output = (float*)malloc(sizeof(float) * numBands);

68 // find the FFT bin index of each freq and store it in bandIndexes

78 void doAnalysis()

82
83 float* windowData = spectrogram->getCurrentWindow();
84
85 int _bandIndex = 1;

87 // finds the total amplitude of each band by summing the bins within each band
88 // then stores that value in output
89 for (int i = this->bandIndexes[0]; i <= this->bandIndexes[this->numBands]; i++) {
90 if (i < this->bandIndexes[_bandIndex]) {
91 _bandSum += windowData[i];
92 } else {
93 this->output[_bandIndex - 1] = _bandSum;
94 _bandIndex += 1;
95 _bandSum = 0;
96 i -= 1;
97 }
98 }
