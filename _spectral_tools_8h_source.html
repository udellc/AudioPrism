129inline float entropy(const float* windowData, int lowerFreq, int upperFreq)
130{
131 if (lowerFreq < 0 || upperFreq > SAMPLE_RATE >> 1 || lowerFreq > upperFreq) {
132 return -1;
133 }
134
135 float freqWidth = (float)WINDOW_SIZE / (float)SAMPLE_RATE;
136 int lowerBinBound = round(lowerFreq * freqWidth);
137 int upperBinBound = round(upperFreq * freqWidth);
138
139 float entropy = 0.;
140 float totalAmp = sum(windowData, lowerBinBound, upperBinBound);
141 for (int i = lowerBinBound; i < upperBinBound; ++i) {
142 // p is the spectral density of the bin, or the percent of the total
143 // amplitude it containts. For calculating entropy, it is helpful to
144 // think of this as the probability of the bin containing energy. The
145 // small offsets are to avoid division by or log of zero errors.
146 float p = (windowData[i] + 1e-12) / (totalAmp + 1e-10);
147 //
148 // Calculate the entropy addition of each bin:
149 //
150 // -log(p) is the 'self-information' of the bin. It is a measure of how
151 // surprising it is to see energy in that bin. If p is close to 1, then
152 // it is not surprising at all to see energy there, but if it is small
153 // it this value will be greater, meaning it is very suprising to see
154 // energy there. This makes sense as p was calculated as the percent of
155 // the total energy that the bin contains.
156 //
157 // We weight this 'self-information' value with the actual percent of
158 // the total amplitude that bin has so that bins with very little energy
159 // do not dominate the entropy value.
160 entropy += p * -(log2(p));
161 }
162
163 // normalize the entropy value by the log2 of the number of bins, which is
164 // the maximum possible entropy value.
165 return entropy / log2(upperBinBound - lowerBinBound);
166}

295inline void smooth_window_over_time(const float* windowData, float* smoothedData,

297{
298 for (int i = 0; i < (WINDOW_SIZE >> 1); ++i) {
