108 float calculate_distance(FormantProfile* profile, float** found_peaks)
109 {
110 // initialize distance to 0
111 // a partial sum is added to distance each iteration of the for loop
112 float distance = 0;
113
114 // calculate partial sum for 5 peaks
115 for (int fpeak = 0; fpeak < NUM_FPEAKS; fpeak++) {
116 // calculate distance between measured peak and profile peak
117 // 1. take the absolute difference in frequency between peaks
118 float d = abs(profile->frequency[fpeak] - found_peaks[0][fpeak]);
119 // 2. if difference is within bandwidth, set distance to 0
120 // d = std::max((float) 0.0, (float) d - profile->bandwidth[fpeak]);
121 // 3. square the distance
122 // if frequency normalization is enabled, increase d to prevent underflows
123 if (FREQUENCY_NORMALIZATION) {
124 d *= 1000;
125 }
126 d *= d;
127
128 for (int i = 0; i < fpeak; i++) {
129 d *= (float)FPEAK_PENALTY;
130 }
131
132 // add partial sum to total distance
133 distance += d;
134 }
135
136 // take square root of the complete sum
137 return (float)pow(distance, 0.5);
138 }
