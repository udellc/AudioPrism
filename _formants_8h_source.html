116 float calculate_distance(FormantProfile *profile, float *found_peaks) {
117 // initialize distance to 0
118 // a partial sum is added to distance each iteration of the for loop
119 float distance = 0;
120
121 // calculate partial sum for 5 peaks
122 for(int fpeak=0; fpeak < num_fpeaks; fpeak++){
123 // calculate distance between measured peak and profile peak
124 // 1. take the absolute difference in frequency between peaks
125
126 float d = abs(profile->frequency[fpeak] - found_peaks[fpeak]);
127 // 2. if difference is within bandwidth, set distance to 0
128 // d = std::max((float) 0.0, (float) d - profile->bandwidth[fpeak]);
129 // 3. square the distance
130 // if frequency normalization is enabled, increase d to prevent underflows
131 if (FREQUENCY_NORMALIZATION) {
132 d *= 1000;
133 }
134 d *= d;
135
136 for (int i = 0; i < fpeak; i++) {
137 d *= (float)FPEAK_PENALTY;
138 }
139
140 // add partial sum to total distance
141 distance += d;
142 }
143
144 // take square root of the complete sum
145 return (float)pow(distance, 0.5);
146 }

148 int interpolateAroundPeak(const float *data, int indexOfPeak, int sampleRate, int windowSize) {
149 float _freqRes = sampleRate * 1.0 / windowSize;
150 float prePeak = indexOfPeak == 0 ? 0.0 : data[indexOfPeak - 1];
151 float atPeak = data[indexOfPeak];
152 float postPeak = indexOfPeak == windowSizeBy2 ? 0.0 : data[indexOfPeak + 1];
153 // summing around the index of maximum amplitude to normalize magnitudeOfChange
154 float peakSum = prePeak + atPeak + postPeak;
155 // interpolating the direction and magnitude of change, and normalizing from -1.0 to 1.0
156 float magnitudeOfChange = ((atPeak + postPeak) - (atPeak + prePeak)) / (peakSum > 0.0 ? peakSum : 1.0);
157
158 // return interpolated frequency
159 return int(round((float(indexOfPeak) + magnitudeOfChange) * _freqRes));
160 }
