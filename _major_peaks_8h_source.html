104

108 outputFrequencies = new float[windowSizeBy2 >> 1];
109 outputAmplitudes = new float[windowSizeBy2 >> 1];
110 }
111
112 // zero out the output arrays
113 for (int i = 0; i < windowSizeBy2 >> 1; i++) {
114 outputFrequencies[i] = 0;
115 outputAmplitudes[i] = 0;
116 }
117 }

119 // find all peaks in the current window
120 // a peak is a freq. bin whose amplitude is greater than its neighbors
121 // this function does not limit itself to finding maxNumPeaks peaks
122 // if more than maxNumPeaks peaks are found, the smallest peaks are removed with trimPeaks later
123 // if fewer than maxNumPeaks peaks are found, the remaining peaks are padded with zeros
124 // this function should only be called after resetPeaksArrays()
125 void findPeaks()
126 {
127 float* windowData = spectrogram->getCurrentWindow();
128 // iterate through the frequency range, excluding the first and last bins
129 for (int i = lowerBinBound + 1; i < upperBinBound; i++) {
130 // if the current bin is a peak, store its frequency and amplitude
131 if (windowData[i] > windowData[i - 1]
132 && windowData[i] > windowData[i + 1]) {
133 // store the frequency of the peak
134 // the index is multiplied by freqRes to convert the bin number to a frequency value
135 // freqRes is the frequency width of each bin
136 outputFrequencies[numPeaks] = i * freqRes;
137
138 // store the amplitude of the peak
139 outputAmplitudes[numPeaks] = windowData[i];
140
141 // increment the number of peaks found to reflect the addition of this peak
142 numPeaks++;
143 }
144 }
145 }
146
147 // remove the smallest peaks until numPeaks <= maxNumPeaks
148 // if there are equal to or fewer than maxNumPeaks peaks, this function does nothing
149 // this function is called directly after findPeaks()
150 // larger peaks are shifted to the left in the temporary storage arrays to overwrite smaller peaks
151 void trimPeaks()
152 {
153 // loop until numPeaks <= maxNumPeaks, removing the smallest peak each iteration
154 while (numPeaks > maxNumPeaks) {
155 float minAmp = outputAmplitudes[0];
156 int minIndex = 0;
157
158 // iterate through the peaks, storing the index of the peak with the smallest amplitude
159 // the peak at this index will be removed in the next step
160 for (int i = 1; i < numPeaks; i++) {
161 if (outputAmplitudes[i] < minAmp) {
162 minAmp = outputAmplitudes[i];
163 minIndex = i;
164 }
165 }
166
167 // remove the peak with the smallest amplitude
168 // removal is done by shifting all elements after the removed peak one index to the left
169 for (int i = minIndex; i < numPeaks - 1; i++) {
170 outputFrequencies[i] = outputFrequencies[i + 1];
171 outputAmplitudes[i] = outputAmplitudes[i + 1];
172 }
173 numPeaks--;
174 }
175 }
176
177 // storePeaks() copies peaks from temporary storage to the actual output arrays
178 // this fuction is called after trimPeaks(), so there can only be up to maxNumPeaks peaks to store
179 // if there are fewer than maxNumPeaks peaks, the remaining elements are padded with zeros
180 void storePeaks()
181 {
182 // copy peaks from temporary storage to the actual output arrays
183 for (int i = 0; i < maxNumPeaks; i++) {
184 // if there are fewer than maxNumPeaks peaks, pad array with zeros
185 if (i < numPeaks) {
186 output[MP_FREQ][i] = outputFrequencies[i];
187 output[MP_AMP][i] = outputAmplitudes[i];
188 } else {
189 output[MP_FREQ][i] = 0;
190 output[MP_AMP][i] = 0;
191 }
192 }
193 }

197 {
198 Serial.printf("MajorPeaks [Freq, Amp]: ");
199 for (int i = 0; i < maxNumPeaks - 1; i++) {
200 Serial.printf("[%03g, %03g], ",

203 }
204 Serial.printf("[%03g, %03g]\n",
205 round(output[MP_FREQ][maxNumPeaks - 1]),
206 round(output[MP_AMP][maxNumPeaks - 1]));
207 }

210 // 1. resetPeaksArrays() to clear the temporary storage and output arrays, and reset the number of peaks found
211 // 2. findPeaks() to find all peaks in the current window (up to windowSizeBy2>>1 peaks)
212 // 3. trimPeaks() to remove the smallest peaks until numPeaks <= maxNumPeaks
213 // 4. storePeaks() to copy peaks from temporary storage to the actual output arrays
214 // this is the function called by the analysis manager to perform the analysis

216 // the output is indexed by peak number, and is always in order of lowest freq peak to highest freq peak
